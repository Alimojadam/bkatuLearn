import React, { useState } from 'react';

const CommentsSection = () => {
  const [comments, setComments] = useState([]);
  const [commentText, setCommentText] = useState("");
  const [repliesOpen, setRepliesOpen] = useState({});
  const [replyText, setReplyText] = useState({});
  const [replyMode, setReplyMode] = useState(null);

  // افزودن کامنت جدید
  const handleAddComment = () => {
    if (!commentText.trim()) return;
    const newComment = {
      id: Date.now(),
      text: commentText,
      replies: []
    };
    setComments([...comments, newComment]);
    setCommentText("");
  };

  // افزودن ریپلای
  const handleAddReply = (parentId) => {
    if (!replyText[parentId]?.trim()) return;
    const updated = comments.map(comment => {
      if (comment.id === parentId) {
        return {
          ...comment,
          replies: [
            ...comment.replies,
            { id: Date.now(), text: replyText[parentId] }
          ]
        };
      }
      return comment;
    });
    setComments(updated);
    setReplyText({ ...replyText, [parentId]: "" });
    setReplyMode(null);
  };

  // ریپلای رو باز یا بسته کن
  const toggleReplies = (id) => {
    setRepliesOpen(prev => ({ ...prev, [id]: !prev[id] }));
  };

  return (
    <div className="w-[80%] mt-10 text-right">
      <h3 className="text-lg font-bold text-[#3073c1] mb-4">نظرات کاربران</h3>

      {/* فرم ارسال نظر جدید */}
      <div className="mb-6">
        <textarea
          className="w-full p-2 border border-gray-300 rounded-md resize-none"
          rows="3"
          placeholder="نظر خود را بنویسید..."
          value={commentText}
          onChange={(e) => setCommentText(e.target.value)}
        />
        <button
          onClick={handleAddComment}
          className="mt-2 px-4 py-1 bg-[#3073c1] text-white rounded-md"
        >
          ارسال نظر
        </button>
      </div>

      {/* لیست نظرات */}
      {comments.map((comment) => (
        <div key={comment.id} className="mb-4 bg-white p-3 rounded-md shadow">
          <p className="mb-2">{comment.text}</p>

          <div className="flex gap-3 text-sm text-[#3073c1] cursor-pointer">
            <span onClick={() => setReplyMode(comment.id)}>پاسخ</span>
            <span onClick={() => toggleReplies(comment.id)}>
              {repliesOpen[comment.id] ? "بستن پاسخ‌ها" : "نمایش پاسخ‌ها"}
            </span>
          </div>

          {/* فرم ریپلای */}
          {replyMode === comment.id && (
            <div className="mt-2">
              <textarea
                rows="2"
                className="w-full p-2 border border-gray-300 rounded-md resize-none"
                placeholder="پاسخ خود را بنویسید..."
                value={replyText[comment.id] || ""}
                onChange={(e) =>
                  setReplyText({ ...replyText, [comment.id]: e.target.value })
                }
              />
              <button
                onClick={() => handleAddReply(comment.id)}
                className="mt-1 px-4 py-1 bg-[#3073c1] text-white rounded-md"
              >
                ارسال پاسخ
              </button>
            </div>
          )}

          {/* ریپلای‌ها */}
          {repliesOpen[comment.id] && comment.replies.length > 0 && (
            <div className="mt-2 border-r-2 border-[#3073c1] pr-3">
              {comment.replies.map((reply) => (
                <div key={reply.id} className="text-sm text-gray-800 py-1">
                  • {reply.text}
                </div>
              ))}
            </div>
          )}
        </div>
      ))}
    </div>
  );
};
